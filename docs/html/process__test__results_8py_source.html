22def load_test_images(test_folder, target_size=(160, 160)):
23 """
24 Load PNG images from a test folder, resize them, and normalize.
25
26 @param test_folder (str): Path to the folder containing test PNG images.
27 @param target_size (tuple): Target size to resize images. Default is (160, 160).
28
29 @return tuple:
30 - numpy.ndarray: Array of preprocessed test images.
31 - list: List of filenames corresponding to the images.
32 """
33 images = []
34 filenames = []
35 for filename in sorted(os.listdir(test_folder)):
36 if filename.endswith('.png'):
37 img_path = os.path.join(test_folder, filename)
38 img = cv2.imread(img_path, cv2.IMREAD_GRAYSCALE)
39 img = cv2.resize(img, target_size)
40 img = img.astype(np.float32) / 255.0
41 images.append(np.expand_dims(img, axis=-1)) # Add channel dimension
42 filenames.append(filename)
43 return np.array(images), filenames

46def process_test_folder(test_folder, model_path, svg_output_dir, target_size=(160, 160)):
47 """
48 Process PNG images from a test folder, predict masks, and save results as SVG.
49
50 @param test_folder (str): Path to the folder containing PNG images.
51 @param model_path (str): Path to the saved model.
52 @param svg_output_dir (str): Directory to save the SVG results.
53 @param target_size (tuple): Target size for resizing images. Default is (160, 160).
54
55 @return None
56 """
57 # Load the trained model
58 model = load_model(model_path)
59
60 # Load test images
61 test_images, filenames = load_test_images(test_folder, target_size)
62
63 # Predict masks
64 pred_masks = model.predict(test_images)
65 pred_masks = (pred_masks > 0.5).astype(np.float32)
66
67 # Use the post-processing module to refine and save predictions as SVG
68 refine_and_save_as_svg(pred_masks, filenames, svg_output_dir, original_images=test_images)
