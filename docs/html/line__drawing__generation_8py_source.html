22def generate_crossing_lines_svg(file_name, num_lines, width=160, height=160):
23 """
24 @brief Generates an SVG file with a specified number of crossing lines.
25
26 @param file_name (str): The name of the SVG file to be saved.
27 @param num_lines (int): The number of crossing lines to generate in the SVG.
28 @param width (int, optional): The width of the SVG canvas in pixels. Default is 160.
29 @param height (int, optional): The height of the SVG canvas in pixels. Default is 160.
30
31 @return None
32 """
33 dwg = svgwrite.Drawing(file_name, size=(width, height))
34
35 possible_stroke_widths = [1]
36
37 for _ in range(num_lines):
38 x1, y1 = random.randint(0, width), random.randint(0, height)
39 x2, y2 = random.randint(0, width), random.randint(0, height)
40 stroke_width = random.choice(possible_stroke_widths)
41 dwg.add(dwg.line(start=(x1, y1), end=(x2, y2), stroke='black', stroke_width=stroke_width))
42
43 dwg.save()
44 print(f"SVG saved as {file_name}")

62 """
63 @brief Converts a PDF file to PNG format.

81def generate_multiple_images(data_folder='data', num_samples_per_line=100, min_lines=3, max_lines=8, width=160, height=160):
82 """
83 @brief Generates multiple SVG, PDF, and PNG images with varying numbers of lines.
84
85 @param data_folder (str, optional): Directory to save generated SVG, PDF, and PNG files. Default is 'data'.

88 @param max_lines (int, optional): Maximum number of lines in SVG images. Default is 8.

90 @param height (int, optional): Height of the SVG and PNG images in pixels. Default is 160.
91
92 @return None

94 if not os.path.exists(data_folder):

102 pdf_file_name = os.path.join(data_folder, f'000_{num_lines}_{line_count:04d}.pdf')
