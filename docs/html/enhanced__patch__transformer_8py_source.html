23class ClipOutput(Layer):
24 """
25 @brief Custom layer to clip model outputs to a specified range.
26
27 @param clip_value_min (float): Minimum value to clip to.
28 @param clip_value_max (float): Maximum value to clip to.
29 """
30 def __init__(self, clip_value_min=0.0, clip_value_max=1.0, **kwargs):
31 super(ClipOutput, self).__init__(**kwargs)
32 self.clip_value_min = clip_value_min
33 self.clip_value_max = clip_value_max
34
35 def call(self, inputs):

39class PositionalEncoding(Layer):
40 """
41 @brief Adds positional encoding to the input patches for better spatial awareness.
42
43 @param num_patches (int): Number of patches in the input.
44 @param d_model (int): Dimension of the embedding space for each patch.
45 """
46 def __init__(self, num_patches, d_model, **kwargs):
47 super(PositionalEncoding, self).__init__(**kwargs)
48 self.num_patches = num_patches
49 self.d_model = d_model
50
51 def call(self, inputs):
52 position = tf.range(self.num_patches, dtype=tf.float32)
53 div_term = tf.exp(tf.range(0., self.d_model, 2.0) * -(tf.math.log(10000.0) / self.d_model))
54 pos_embedding = tf.matmul(tf.expand_dims(position, -1), div_term[tf.newaxis, :])
55 pos_embedding = tf.concat([tf.sin(pos_embedding), tf.cos(pos_embedding)], axis=-1)
56 return inputs + tf.expand_dims(pos_embedding, 0)

82def transformer_encoder(inputs, num_heads, key_dim, ff_dim, dropout_rate=0.1):
83 """
84 @brief Builds a single Transformer encoder block.
85
86 @param inputs (Tensor): Input tensor to the encoder block.
87 @param num_heads (int): Number of attention heads.
88 @param key_dim (int): Dimensionality of the attention key space.
89 @param ff_dim (int): Dimensionality of the feedforward network.
90 @param dropout_rate (float, optional): Dropout rate for regularization. Default is 0.1.
91
92 @return Tensor: Output tensor from the Transformer encoder block.
93 """
94 attention_output = tf.keras.layers.MultiHeadAttention(num_heads=num_heads, key_dim=key_dim)(inputs, inputs)
95 attention_output = Add()([inputs, attention_output])
96 attention_output = LayerNormalization()(attention_output)
97
98 ff_output = Dense(ff_dim, activation="relu")(attention_output)
99 ff_output = Dense(inputs.shape[-1], activation="relu")(ff_output)
100 ff_output = Add()([attention_output, ff_output])
101 ff_output = LayerNormalization()(ff_output)
102
103 return Dropout(dropout_rate)(ff_output)

106def create_enhanced_patch_transformer_model(img_size=160, patch_size=40, stride=40):
107 """
108 @brief Creates the Enhanced Patch Transformer model for line segmentation.
109
110 @param img_size (int, optional): Input image size (assumed square). Default is 160.
111 @param patch_size (int, optional): Size of each patch. Default is 40.
112 @param stride (int, optional): Stride for extracting patches. Default is 40.
113
114 @return Model: Keras Model instance of the Enhanced Patch Transformer.
115 """
116 num_patches = (img_size // patch_size) ** 2
117 image_input = Input(shape=(img_size, img_size, 1), name="input_layer")
118 patches = PatchExtractor(patch_size=patch_size, stride=stride)(image_input)
119 patches_reshaped = Reshape((-1, patch_size, patch_size, 1))(patches)
120
121 # Patch Embedding via CNN Layers
122 x = TimeDistributed(Conv2D(32, (3, 3), activation='relu', padding='same'))(patches_reshaped)
123 x = TimeDistributed(Conv2D(64, (3, 3), activation='relu', padding='same'))(x)
124 x = TimeDistributed(Flatten())(x)
125 patch_features = TimeDistributed(Dense(128, activation='relu'))(x)
126 patch_grid = Reshape((num_patches, 128))(patch_features)
127
128 # Positional Encoding
129 patch_grid = PositionalEncoding(num_patches, 128)(patch_grid)
130
131 # Stacked Transformer Encoders
132 for _ in range(3):
133 patch_grid = transformer_encoder(patch_grid, num_heads=2, key_dim=32, ff_dim=128, dropout_rate=0.2)
134
135 # Global Aggregation
136 global_avg = GlobalAveragePooling1D()(patch_grid)
137 global_max = GlobalMaxPooling1D()(patch_grid)
138 global_features = Concatenate()([global_avg, global_max])
139
140 # Output Coordinates
141 line_coordinates = Dense(8 * 4, activation='linear', name="line_coordinates")(global_features)
142 line_coordinates = Reshape((8, 4), name="final_output")(line_coordinates)
143
144 # Clip output to [0, 1]
145 clipped_output = ClipOutput(clip_value_min=0.0, clip_value_max=1.0)(line_coordinates)
146
147 return Model(inputs=image_input, outputs=clipped_output)

150def scaled_loss(y_true, y_pred):
151 """
152 @brief Computes the custom scaled loss for line prediction.
153
154 @param y_true (Tensor): Ground truth line coordinates.
155 @param y_pred (Tensor): Predicted line coordinates.
156
157 @return Tensor: Scaled loss value.
158 """
159 true_coords = y_true[:, :, :4]
160 pred_coords = y_pred[:, :, :4]
161 coord_loss = tf.reduce_mean(tf.reduce_sum(tf.square(pred_coords - true_coords), axis=-1))
162 return coord_loss
