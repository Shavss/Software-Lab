@file enhanced_patch_transformer.py
@brief Defines the Enhanced Patch Transformer model for line segmentation.

@details
This module includes:
- Custom layers (PositionalEncoding, PatchExtractor, ClipOutput)
- Transformer encoder architecture
- Enhanced Patch Transformer model for processing image patches.

108 @brief Creates the Enhanced Patch Transformer model for line segmentation.

110 @param img_size (int, optional): Input image size (assumed square). Default is 160.
111 @param patch_size (int, optional): Size of each patch. Default is 40.
112 @param stride (int, optional): Stride for extracting patches. Default is 40.
113
114 @return Model: Keras Model instance of the Enhanced Patch Transformer.
115 """

117 image_input = Input(shape=(img_size, img_size, 1), name="input_layer")
118 patches = PatchExtractor(patch_size=patch_size, stride=stride)(image_input)
119 patches_reshaped = Reshape((-1, patch_size, patch_size, 1))(patches)
120
121 # Patch Embedding via CNN Layers
122 x = TimeDistributed(Conv2D(32, (3, 3), activation='relu', padding='same'))(patches_reshaped)
123 x = TimeDistributed(Conv2D(64, (3, 3), activation='relu', padding='same'))(x)
124 x = TimeDistributed(Flatten())(x)
125 patch_features = TimeDistributed(Dense(128, activation='relu'))(x)
126 patch_grid = Reshape((num_patches, 128))(patch_features)
127
128 # Positional Encoding
129 patch_grid = PositionalEncoding(num_patches, 128)(patch_grid)
130
131 # Stacked Transformer Encoders
132 for _ in range(3):
133 patch_grid = transformer_encoder(patch_grid, num_heads=2, key_dim=32, ff_dim=128, dropout_rate=0.2)
134
135 # Global Aggregation
136 global_avg = GlobalAveragePooling1D()(patch_grid)
137 global_max = GlobalMaxPooling1D()(patch_grid)
138 global_features = Concatenate()([global_avg, global_max])
139
140 # Output Coordinates
141 line_coordinates = Dense(8 * 4, activation='linear', name="line_coordinates")(global_features)
142 line_coordinates = Reshape((8, 4), name="final_output")(line_coordinates)
143
144 # Clip output to [0, 1]
145 clipped_output = ClipOutput(clip_value_min=0.0, clip_value_max=1.0)(line_coordinates)
