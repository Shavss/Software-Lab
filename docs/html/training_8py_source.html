35def get_callbacks(model_checkpoint_path='models/best_model.keras'):
36 """
37 @brief Creates callbacks for training.
38
39 This function sets up EarlyStopping to halt training when validation loss stops improving
40 and ModelCheckpoint to save the best model during training.
41
42 @param model_checkpoint_path (str, optional): Path to save the best model. Default is 'models/best_model.keras'.
43
44 @return list: List of callback instances.
45 """
46 earlystopper = EarlyStopping(
47 monitor='val_loss',
48 patience=10,
49 verbose=1,
50 restore_best_weights=True
51 )
52 checkpointer = ModelCheckpoint(
53 model_checkpoint_path,
54 monitor='val_loss',
55 verbose=1,
56 save_best_only=True
57 )
58 return [earlystopper, checkpointer]

61def train_model(model, train_gen, val_gen, X_train, X_val, batch_size=16, epochs=100, model_name="model"):
62 """
63 @brief Trains a Keras model using the provided generators.
64
65 This function fits the model on the training data using data generators for augmentation,
66 validates on the validation data, and applies callbacks for early stopping and model checkpointing.
67
68 @param model (Model): Compiled Keras model to train.
69 @param train_gen (generator): Training data generator.
70 @param val_gen (generator): Validation data generator.
71 @param X_train (numpy.ndarray): Number of training samples.
72 @param X_val (numpy.ndarray): Number of validation samples.
73 @param batch_size (int, optional): Number of samples per batch. Default is 16.
74 @param epochs (int, optional): Maximum number of training epochs. Default is 100.
75 @param model_name (str, optional): Name of the model for saving the best weights. Default is "model".
76
77 @return History: Keras History object containing training history.
78 """
79 model_checkpoint_path = f'models/{model_name}_best_model.keras'
80 callbacks = get_callbacks(model_checkpoint_path)
81
82 history = model.fit(
83 train_gen,
84 steps_per_epoch=len(X_train) // batch_size,
85 epochs=epochs,
86 validation_data=val_gen,
87 validation_steps=len(X_val) // batch_size,
88 callbacks=callbacks
89 )
90 return history
